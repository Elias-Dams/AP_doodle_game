149bool World::colisionCheck(shared_ptr<Entity_Model> entity, bool can_update_score) {
150 // the left side of the player
151 if (entity->getPosition().first <= player->getPosition().first and
152 player->getPosition().first <= entity->getPosition().first + entity->getWidth()) {
153
154 if (player->getPosition().second - (entity->getPosition().second + entity->getHeight()) <= 0 and
155 player->getPosition().second - entity->getPosition().second >= 0 and player->isfalling()) {
156
157 if(can_update_score){
158 player->setPlatfomTouched(entity);
159 }
160
161 return true;
162 }
163 }
164 // right side of the player
165 else if (entity->getPosition().first <= player->getPosition().first + player->getWidth() and
166 player->getPosition().first + player->getWidth() <= entity->getPosition().first + entity->getWidth()) {
167
168 if (player->getPosition().second - (entity->getPosition().second + entity->getHeight()) <= 0 and
169 player->getPosition().second - entity->getPosition().second >= 0 and player->isfalling()) {
170
171 if(can_update_score){
172 player->setPlatfomTouched(entity);
173 }
174
175 return true;
176 }
177 }
178
179 // between the player
180 else if (entity->getPosition().first >= player->getPosition().first and
181 player->getPosition().first + player->getWidth() >= entity->getPosition().first + entity->getWidth()) {
182
183 if (player->getPosition().second - (entity->getPosition().second + entity->getHeight()) <= 0 and
184 player->getPosition().second - entity->getPosition().second >= 0 and player->isfalling()) {
185

190 return true;

192 }
193 return false;

196void World::startstate(float dt) {
197 if (platforms.empty()) {
198 this->create_start_platform();
199 player->PlayerReset(50.0f, 300.0f);

202 bool hit = false;
203 vector<shared_ptr<Platform>>::iterator it = platforms.begin();
204 while (it != platforms.end()) {
205
206 (*it)->update(dt);
207
208 // false because we can't alter the score

218void World::update(float dt, const char &key) {
219 if (platforms.size() == 1) {
220 this->create_platforms();
221 }
222 // game over;
223 if (player->getPosition().second < (camera->getNomalisedHeight() - HEIGHT / 2)) {
224 GameOver = true;
225 }
226
227 // update the camera pos
228 if (player->getPosition().second >= camera->getNomalisedHeight()) {
229 camera->setHeight(player->getPosition().second);
230 }
231
232 // update the background
233 if (background.second->getPosition().second + HEIGHT / 2 <= camera->getNomalisedHeight()) {
234 background.first->setPosition(background.second->getPosition().first,
235 background.second->getPosition().second + HEIGHT);
236
237 shared_ptr<BG_Tile> temp = background.second;
238 background.second = background.first;
239 background.first = temp;
240 }
241
242 // move the player
243 if (key == 'Q') {
244 player->move(-5.f * dt * 60.f, 0.f * dt * 60.f);
245 }
246 if (key == 'D') {
247 player->move(5.f * dt * 60.f, 0.f * dt * 60.f);
248 }
249
250 // if player goes out of the screen to the left it wil go to the right.
251 if (player->getPosition().first <= -player->getWidth()) {
252 player->setPosition(WIDTH - player->getWidth(), player->getPosition().second);
253 }
254 // if player goes out of the screen to the right it wil go to the left.
255 if (player->getPosition().first >= WIDTH) {
256 player->setPosition(0, player->getPosition().second);
257 }
258
259 // check collision
260 int height_modification = 1;
261
262 bool hit = false;
263 bool bonus_hit = false;
264
265 vector<shared_ptr<Platform>>::iterator it = platforms.begin();
266 while (it != platforms.end()) {
267 bool current_hit = false;
268
269 (*it)->update(dt);
270
271 if (bonusses.count((*it))) {
272 bonusses[(*it)]->update(
273 (*it)->getPosition().first + (*it)->getWidth() / 2 - bonusses[(*it)]->getWidth() / 2,
274 (*it)->getPosition().second + (*it)->getHeight());
275 // true because we can alter the score
276 if (this->colisionCheck(bonusses[(*it)], true)) {
277 hit = true;
278 bonus_hit = true;
279 current_hit = true;
280 height_modification = bonusses[(*it)]->getBonuspower();
281 }
282 }
283 // true because we can alter the score
284 if (this->colisionCheck(*it, true) and !hit) {
285 hit = true;
286 current_hit = true;
287 }
288
289 if ((*it)->getPosition().second < (camera->getNomalisedHeight() - HEIGHT / 2) - (*it)->getHeight()) {
290 if (bonusses.count((*it))) {
291 factory->delete_bonus(bonusses[(*it)]);
292 bonusses[(*it)].reset();
293 bonusses.erase((*it));
294 }
295 factory->delete_platform((*it));
296 (*it).reset();
297 platforms.erase(it);
298
299 } else if (current_hit and (*it)->getColor() == "White") {
300 factory->delete_platform((*it));
301 (*it).reset();
302 platforms.erase(it);
303 } else if (current_hit and bonus_hit and bonusses.count((*it))) {
304 // make the jetpack disappear
305 if (bonusses[(*it)]->getBonuspower() == 12) {
306 factory->delete_bonus(bonusses[(*it)]);
307 bonusses[(*it)].reset();
308 bonusses.erase((*it));
309 } else {
310 ++it;
311 }
312 } else {
313 ++it;
314 }
315 }
316
317 // if we het higher there need to be less platforms
318 // because of the way draw my platforms the constant has to be smaller or equil to
319 // the height if the world * 2 /100 <= platforms per 2 views - constant
320 // |---------600 * 2 / 100--------| <= |--------20 - constant------------| ==> 12 <= 20 - constant
321 // if the constant becomes more than it is almost impossible to draw platforms that are reachable bij the player
322 if ((int) camera->getNomalisedHeight() >= 3000 * constant && constant <= 16) {
323 constant += 1;
324 platforms_per_view -= 1;
325 }
326
327 // add platforms
328 this->add_platforms();
329
330 player->jump(dt, hit, height_modification);
331

359void World::Reset() {
